# Doa_Roupa_Flutter_projeto1.0 - sem supabase
para criar e testar seu próprio rode no supabase para replicar as tabelas:
CREATE TABLE public.atividades ( id uuid NOT NULL DEFAULT uuid_generate_v4(), tipo text NOT NULL, titulo text NOT NULL, descricao text NOT NULL, itens jsonb NOT NULL, data_inicio date NOT NULL, data_fim date NOT NULL, status text DEFAULT 'em andamento'::text, created_at timestamp without time zone DEFAULT now(), CONSTRAINT atividades_pkey PRIMARY KEY (id) ); CREATE TABLE public.doacoes ( id uuid NOT NULL DEFAULT uuid_generate_v4(), doador_id uuid, atividade_id uuid, tipo_roupa text NOT NULL, genero text NOT NULL, tamanho text NOT NULL, quantidade integer NOT NULL, anonimo boolean DEFAULT false, status text DEFAULT 'pendente'::text, created_at timestamp without time zone DEFAULT now(), CONSTRAINT doacoes_pkey PRIMARY KEY (id), CONSTRAINT doacoes_atividade_id_fkey FOREIGN KEY (atividade_id) REFERENCES public.atividades(id), CONSTRAINT doacoes_doador_id_fkey FOREIGN KEY (doador_id) REFERENCES public.usuarios(id) ); CREATE TABLE public.estoque ( id uuid NOT NULL DEFAULT uuid_generate_v4(), tipo_roupa text NOT NULL, genero text NOT NULL, tamanho text NOT NULL, quantidade integer NOT NULL, created_at timestamp without time zone DEFAULT now(), CONSTRAINT estoque_pkey PRIMARY KEY (id) ); CREATE TABLE public.usuarios ( id uuid NOT NULL, nome text NOT NULL, email text NOT NULL UNIQUE, papel text NOT NULL CHECK (papel = ANY (ARRAY['doador'::text, 'admin'::text])), genero text, idade integer, profile_url text, mudanca_cargo text DEFAULT 'não solicitado'::text, CONSTRAINT usuarios_pkey PRIMARY KEY (id), CONSTRAINT usuarios_id_fkey FOREIGN KEY (id) REFERENCES auth.users(id) );
